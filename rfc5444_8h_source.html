95 static INLINE bool
96 rfc5444_seqno_is_larger(uint16_t s1, uint16_t s2) {
97  /*
98  * The sequence number S1 is said to be "greater than" the sequence
99  * number S2 if:
100  * o S1 > S2 AND S1 - S2 < MAXVALUE/2 OR
101  * o S2 > S1 AND S2 - S1 > MAXVALUE/2
102  */
103  return (s1 > s2 && (s1 - s2) < (1 << 15)) || (s2 > s1 && (s2 - s1) > (1 << 15));
104 }

112 static INLINE bool
113 rfc5444_seqno_is_smaller(uint16_t s1, uint16_t s2) {
114  return s1 != s2 && !rfc5444_seqno_is_larger(s1, s2);
