172 #define _OONF_LOG(severity, source, hexptr, hexlen, condition, fail, format, args...) \
173  do { \
174  if ((condition) && oonf_log_mask_test(log_global_mask, source, severity)) { \
175  oonf_log(severity, source, &__FILE__[BASEPATH_LENGTH], __LINE__, hexptr, hexlen, format, ##args); \
176  if (OONF_DO_ABORT && (fail)) { abort(); } \
177  } \
178  } while (0)
