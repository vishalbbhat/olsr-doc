504 /*
505  * Example of a section schema definition with binary mapping
506  *
507  * All CFG_MAP_xxx macros follow a similar pattern.
508  * - the first parameter is the name of the struct the data will be mapped into
509  * - the second parameter is the name of the field the data will be mapped into
510  * - the third parameter is the name of the key in the configuration file
511  * - the fourth parameter is the default value (as a string!)
512  * - the fifth parameter is the help text
513  *
514  * struct bin_data {
515  * char *string;
516  * int int_value;
517  * };
518  *
519  * static struct cfg_schema_section section =
520  * {
521  * .type = "testsection",
522  * .mode = CFG_SSMODE_NAMED
523  * };
524  *
525  * static struct cfg_schema_entry entries[] = {
526  * CFG_MAP_PRINTABLE(bin_data, string, "text", "defaulttext", "help for text parameter"),
527  * CFG_MAP_INT_MINMAX(bin_data, int_value, "number", "0", "help for number parameter", 0, 10),
528  * };
